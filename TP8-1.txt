DROP DATABASE IF EXISTS voyage_moldovan_miyuki;

CREATE DATABASE voyage_moldovan_miyuki;

use voyage_moldovan_miyuki;

SET FOREIGN_KEY_CHECKS = 0;
DROP TABLE IF EXISTS `Pays`;
DROP TABLE IF EXISTS `Ville`;
DROP TABLE IF EXISTS `Adresse`;
DROP TABLE IF EXISTS `Trajet`;
DROP TABLE IF EXISTS `Companie`;
DROP TABLE IF EXISTS `Conducteur`;
DROP TABLE IF EXISTS `Transport`;
DROP TABLE IF EXISTS `TypeTransport`;
DROP TABLE IF EXISTS `Personne`;
DROP TABLE IF EXISTS `Voyageur`;
DROP TABLE IF EXISTS `CTT`;
DROP TABLE IF EXISTS `Voyage`;
DROP TABLE IF EXISTS `Agence`;
DROP TABLE IF EXISTS `AdresseVille`;
DROP TABLE IF EXISTS `VV`;
DROP TABLE IF EXISTS `VT`;
SET FOREIGN_KEY_CHECKS = 1;

CREATE TABLE `Pays` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE (`name`)
);

CREATE TABLE `Ville` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(255) NOT NULL,
    `PaysId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE (`name`)
);

CREATE TABLE `Adresse` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `label` TEXT NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `Trajet` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `date_depart` DATE NOT NULL,
    `date_arrive` DATE NOT NULL,
    `AdresseDepartId` INTEGER UNSIGNED NOT NULL,
    `AdresseArriveId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `Companie` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(255) NOT NULL,
    `siegeId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `Conducteur` (
    `id` INTEGER UNSIGNED NOT NULL,
    `guid` VARCHAR(255) NOT NULL,
    `companyId` INTEGER UNSIGNED NOT NULL,
    `date_debut` DATE NOT NULL,
    `date_fin` DATE NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE (`guid`)
);

CREATE TABLE `Transport` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `typeTransportId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `TypeTransport` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `label` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `Personne` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `firstname` VARCHAR(255) NOT NULL,
    `lastname` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `Voyageur` (
    `id` INTEGER UNSIGNED NOT NULL,
    `dossier` VARCHAR(30) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE (`dossier`)
);

CREATE TABLE `CTT` (
    `conducteurId` INTEGER UNSIGNED NOT NULL,
    `transportId` INTEGER UNSIGNED NOT NULL,
    `trajetId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`conducteurId`, `transportId`, `trajetId`)
);

CREATE TABLE `Voyage` (
    `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE (`name`)
);

CREATE TABLE `Agence` (
    `CompanieId` INTEGER UNSIGNED NOT NULL,
    `AdresseId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`CompanieId`, `AdresseId`)
);

CREATE TABLE `AdresseVille` (
    `VilleId` INTEGER UNSIGNED NOT NULL,
    `AdresseId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`VilleId`, `AdresseId`)
);

CREATE TABLE `VV` (
    `VoyageurId` INTEGER UNSIGNED NOT NULL,
    `voyageId` INTEGER UNSIGNED NOT NULL,
    `prix` DECIMAL NOT NULL,
    `notation` INTEGER UNSIGNED,
    PRIMARY KEY (`VoyageurId`, `voyageId`)
);

CREATE TABLE `VT` (
    `voyageId` INTEGER UNSIGNED NOT NULL,
    `trajetId` INTEGER UNSIGNED NOT NULL,
    PRIMARY KEY (`voyageId`, `trajetId`)
);

ALTER TABLE `Ville` ADD FOREIGN KEY FK_Ville_PaysId_Pays_id (`PaysId`) REFERENCES `Pays`(`id`);
